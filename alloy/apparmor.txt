#include <tunables/global>

profile alloy flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6 Overlay and initialization.
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /usr/local/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /run/{,**} rwk,
  /tmp/** rwk,
  /data/** rw,
  /etc/alloy/** rw,

  /usr/bin/alloy cx -> alloy-service,

  profile alloy-service flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    signal (receive) peer=*_alloy,
    capability dac_override,
    capability dac_read_search,

    network inet stream,
    network inet6 stream,
    network inet dgram,
    network inet6 dgram,
    network unix stream,
    network netlink raw,

    /usr/bin/alloy r,
    /etc/alloy/{,**} r,
    /data/alloy/{,**} rwk,
    /var/log/journal/{,**} r,
    /run/log/journal/{,**} r,
    /etc/machine-id r,
    /proc/{,**} r,
    /sys/fs/cgroup/{,**} r,
    /tmp/{,**} rwk,
    /dev/null rw,
    /dev/urandom r,
  }
}
